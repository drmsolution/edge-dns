To get the LoadBalancer IP:
  kubectl -n {{ include "edge-dns.namespace" . }} get svc {{ include "edge-dns.componentName" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}'

To port-forward the Admin API locally:
  kubectl -n {{ include "edge-dns.namespace" . }} port-forward svc/{{ include "admin-api.name" . }} {{ .Values.admin-api.port }}:{{ .Values.admin-api.port }}
  curl http://localhost:{{ .Values.admin-api.port }}/api/v1/analytics/summary
